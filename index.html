JavaScript String Methods

let txt = "Hello World!"

  • length
    console.log(txt.length);
    outoput: 12
  • slice()
    console.log(txt.slice(2,8));
    output: llo Wor
    console.log(txt.slice(1));
    output: ello World!
    console.log(txt.slice(-6));
    output: World!
  • toUpperCase()
    console.log(txt.toUpperCase());
    output: HELLO WORLD!
  • toLowerCase()
    console.log(txt.toLowerCase());
    output: hello world!
  • Replacing string content
    console.log(txt.replace("World", "Vietnam!"));
    output: Hello Vietnam!
  • concat()
    let txt2 = "Wassup-"
    let txt3 = txt2.concat(txt);
    console.log(txt3);
    output = Wassup-Hello World!
